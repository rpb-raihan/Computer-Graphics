{
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3d(1,0,0);
    glLoadIdentity(); //Reset the drawing perspective
    glMatrixMode(GL_MODELVIEW);

    glPushMatrix();
    glTranslatef(0.3,0.1,0.0);
    glRotatef(_angle1, 0.0f, 0.0f,1.0f);
    glTranslatef(-0.3,-0.1,0.0);
    glBegin(GL_QUADS);
    glVertex2f(0.1f, 0.0f);
    glVertex2f(0.5f, 0.0f);
    glVertex2f(0.5f, 0.2f);
    glVertex2f(0.1f, 0.2);
    glEnd();
    glPopMatrix();
    glutSwapBuffers();
}

void update(int value)
{

    _angle1+=2.0f;
    glutPostRedisplay(); //Notify GLUT that the display has change
    glutTimerFunc(20, update, 0); //Notify GLUT to call update again in 25 milliseconds
}

    glutMainLoop();

    glEnd();
    glPopMatrix();

void drawWindmill() {
    // Draw windmill base
    glColor3d(0.5, 0.5, 0.5);
    glBegin(GL_QUADS);
    glVertex2f(-0.1f, -2.0f);
    glVertex2f(0.1f, -2.0f);
    glVertex2f(0.1f, 0.0f);
    glVertex2f(-0.1f, 0.0f);
    glEnd();

    // Draw blades
    for (int i = 0; i < 4; ++i) {
        glPushMatrix();
        glRotatef(i * 90.0f, 0.0f, 0.0f, 1.0f);
        drawBlade();
        glPopMatrix();
    }
}
